Churchill

This is a sample of an analysis of Winston Churchills We Shall Fight speech. The text was taken from: https://winstonchurchill.org/resources/speeches/1940-the-finest-hour/we-shall-fight-on-the-beaches/

I created this in RStudio after reading a tutorial found here: http://www.sthda.com/english/wiki/text-mining-and-word-cloud-fundamentals-in-r-5-simple-steps-you-should-know

# Load
library("tm")
Loading required package: NLP
Warning message:
R graphics engine version 12 is not supported by this version of RStudio. The Plots tab will be disabled until a newer version of RStudio is installed. 
library("SnowballC")
library("wordcloud")
Loading required package: RColorBrewer
library("RColorBrewer")
filePath <-"Churchill.txt"
docs <- Corpus(VectorSource(text))
toSpace <- content_transformer(function (x , pattern ) gsub(pattern, " ", x))
docs <- tm_map(docs, toSpace, "/")
transformation drops documents
docs <- tm_map(docs, toSpace, "@")
transformation drops documents
docs <- tm_map(docs, toSpace, "\\|")
transformation drops documents
# Convert the text to lower case, remove numbers, remove filler words, remove punctuation and white space
docs <- tm_map(docs, content_transformer(tolower))
transformation drops documents
docs <- tm_map(docs, removeNumbers)
transformation drops documents
docs <- tm_map(docs, removeWords, stopwords("english"))
transformation drops documents
docs <- tm_map(docs, removeWords, c("upon", "shall")) 
transformation drops documents
docs <- tm_map(docs, removePunctuation)
transformation drops documents
docs <- tm_map(docs, stripWhitespace)
transformation drops documents
# Text stemming (simplifying words to their root)
#docs <- tm_map(docs, stemDocument)
dtm <- TermDocumentMatrix(docs)
m <- as.matrix(dtm)
v <- sort(rowSums(m),decreasing=TRUE)
d <- data.frame(word = names(v),freq=v)
head(d, 10)
set.seed(1234)
wordcloud(words = d$word, freq = d$freq, min.freq = 1,
          max.words=74, random.order=FALSE, rot.per=0.35, 
          colors=brewer.pal(8, "Dark2"))
